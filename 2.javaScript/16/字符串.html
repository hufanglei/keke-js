4.正则表达式
两种形态 
第一种:字符串形态 ： "/^正则表达式$/" ---常量
第二种:对象形态: RegExp regex = new Regex("正则表达式",匹配模式(igm));
一般用于:变量形态对的
i代表忽略大小写
g代表是全局
m是多行匹配
//replace:

\d  数字
\D  非数字
\w  数字，字母，下划线
\W  非数字字母，下划线
*  0或多个
?  单个
+  一个或多个
.  代表单个字符
[a-zA-Z]  必须有字母，包含数字，特殊字符
{8} 必须是8位
{8,} num>=8
[11||12||13] 范围的意思

var r = /^(\d{4})-(\d{1,2})-(\d{1,2})$/;
r.exec('1985-10-15');
s1=RegExp.$1;
s2=RegExp.$2;
s3=RegExp.$3;
console.log(s1+"==="+s2+"==="+s3);//1985===10===15

var r = /^(\d{4})-(\d{1,2})-(\d{1,2})$/;
var c = "1985-10-15".replace(r,"$1#$2#$3");
console.log(c);//1985#10#15

var str = "我123可可456";
var s = str.replace(/^.*?(\d{3}).*?(\d{3})$/,"$1$2");
var s1 = str.replace(/^.*?(\d{3}).*?$/,"$1");
console.log(s);//123456
console.log(s1);//123
