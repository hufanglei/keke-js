拼音转汉字

实现办法:
①引入js:
<script type="text/javascript" src="pinyin.js"></script>
二次封装代码:
var Pinyin = {
	get:function(content,mark){
		var py = CC2PY(content);
		return mark?py:py.replace(/\s*/g,"");//是否去空格显示，true空格，false:不空格
	}
};

//②页面代码：
<textarea onkeyup="enterChinese(this)"></textarea> 
<textarea id="box"></textarea> 

//③js代码:
function enterChinese(obj){
	var content = Pinyin.get(obj.value,false);//false代表去空格，true代表不去空格
	document.getElementById("box").innerHTML = content;
}
