//html代码:
<canvas id="canvas" width="960" height="460" style="background:#111;"></canvas>

//JS代码:
<script>
	window.onload = function(){
		//使用 id 来寻找 canvas 元素：
		var canvas = document.getElementById("canvas");
		//创建 context 对象：
		var context = canvas.getContext("2d");

		//画圆
		context.beginPath();
		context.fillStyle = "red";
		context.arc(200,200,50,0,Math.PI*2);
		context.fill();

		//画矩形
		context.beginPath();
		context.fillRect(300,300,200,100);
		context.fill();

		//绘制文字
		context.beginPath();
		context.fillStyle = "green";
		context.font = "18px 微软雅黑 ";
		context.fillText("我爱你！",50,50);
		context.fill();

		//绘制线条
		context.beginPath();
		context.strokeStyle = "green";
		context.moveTo(100,100);
		context.lineTo(250,50);
		context.lineTo(10,90);
		context.stroke();
	}
	
</script>
