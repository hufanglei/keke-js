案例1:

开启 关闭





实现代码: 
//html代码:
<div id="cox"></div>
<a href="javascript:void(0)" onclick="play()">开启</a>
<a href="javascript:void(0)" onclick="cclose()">关闭</a>
//js代码:
var num=0;
var ctop = 0;
var left=0;
var speed =3;
function change3(){
	num++;
	ctop+=speed;
	left+=speed;
	//document.getElementById("num").innerHTML = num;
	document.getElementById("cox").style.left = left + "px";
	document.getElementById("cox").style.top = ctop + "px";
	speed+=0.1;
};
var timer3 = null;
function play(){
	timer3 = setInterval(change3,1000/30);
}
function cclose(){
	clearInterval(timer3);
}
