随机数案例


案例1:范围随机
//js代码:
function rangeRandom(start ,end){
	return Math.floor(Math.random()*(end-start+1)+start);
}
setInterval(function(){
	console.log(rangeRandom(8,10));//运行结果:8,9,10随机生成
},1000);

案例2:单个随机
//js代码:
function tRandom(start){
	return Math.floor(Math.random()*start);
}
setInterval(function(){
	console.log(tRandom(10));//运行结果:0-10随机生成
},1000);

案例3:RGB颜色随机
//js代码:
function getRgb(){
	//0-255
	var r = tRandom(256);
	var g = tRandom(256);
	var b = tRandom(256);
	return "rgb("+r+","+g+","+b+")";
};
setInterval(function(){
	console.log(getRgb());
},1000);

案例4:16进制颜色随机
//js代码:
function getRgb16(){
		//0-255
		var r = tRandom(256).toString(16);
		var g = tRandom(256).toString(16);
		var b = tRandom(256).toString(16);
		if(r.length<2)r=r+""+r;
		if(g.length<2)g=g+""+g;
		if(b.length<2)b=b+""+b;
		return "#"+r+""+g+""+b;
	}
setInterval(function(){
	console.log(getRgb16());
},1000);

案例5:色卡随机生成
//html代码:
<button onclick="start();">启动</button>
<button onclick="toEnd();">暂停</button><br/>
//js代码:
function getRgb16(){
	//0-255
	var r = tRandom(256).toString(16);
	var g = tRandom(256).toString(16);
	var b = tRandom(256).toString(16);
	if(r.length<2)r=r+""+r;
	if(g.length<2)g=g+""+g;
	if(b.length<2)b=b+""+b;
	return "#"+r+""+g+""+b;
}
setInterval(function(){
	console.log(getRgb16());
},1000);
var timer = null;
function createSpan(){
	var spanDom = document.createElement("span");
	var color = getRgb16();
	spanDom.style.background = color;
	spanDom.innerHTML = color;
	document.body.appendChild(spanDom);
};
function start(){
	if(timer)clearInterval(timer);
	timer = setInterval(createSpan,1000);
};

function toEnd(){
	if(timer)clearInterval(timer);
};
