  • 选项卡的制作:jquery版本
	//html
	<div id="tab">
		<ul id="card">
			<li >选项卡1</li>
			<li >选项卡2</li>
			<li >选项卡3</li>		
			<li >选项卡4</li>		
		</ul>
		<div id="content" class="content">
			<div  >内容1</div>
			<div  >内容2</div>
			<div  >内容3</div>
			<div  >内容4</div>
		</div>
	</div>	

	//js:
	$("#card").children().click(function(){		
		var $li = $(this);//获取当前的li		
		var index = $li.index();	//获取点击的索引		
		$li.addClass("on").siblings().removeClass("on");//添加旋转样式，同辈互斥		
		$("#content").children().eq(index).show().siblings().hide();//内容显示，同辈互斥
	});
