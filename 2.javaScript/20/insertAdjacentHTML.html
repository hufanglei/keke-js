		function kekebefrore(dom ,html){
			dom.insertAdjacentHTML("beforebegin",html);
		}
		function kekeAfeter(dom ,html){
			dom.insertAdjacentHTML("afterbegin",html);
		}
		kekebefrore(boxDom,"<strong>你好</strong>");
		kekeAfeter(boxDom,"<strong>keke</strong>");
		kekebefrore(boxDom,"<strong>过得怎么样</strong>");
		kekeAfeter(boxDom,"<strong>很好</strong>");
