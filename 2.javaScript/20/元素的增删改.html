1:元素的增加

①appendChild(dom对象)

		//①appendChild(dom对象)
		ar spanDom = document.createElement("div");
		spanDom.innerHTML ="我是一个动态span奥";			
		$("box").appendChild(spanDom);

②insertBefore(追加dom子元素，在哪个元素前追加)

		$("box").insertBefore(spanDom,$("box").firstElementChild);
		//插入到哪个位置
		$("box").insertBefore(spanDom,$("box").children[1]);

2:元素的删除

removeChild(dom对象) ，没有直接删除的概念，在javascript中一定是和父亲脱离关系

		var spanDom = document.createElement("div");
		spanDom.innerHTML ="0000";		
		var aDom = document.createElement("a");
		aDom.innerHTML = "删除";
		aDom.href = "javascript:void(0)";
		aDom.onclick = function(){
			$("box").removeChild(this.parentElement);
		}
		spanDom.appendChild(aDom);
		$("box").appendChild(spanDom);

3:元素的替换

replaceChild(新替换dom,被替换dom)

		var spanDom2 = document.createElement("strong");
		spanDom2.innerText = spanDom2.textContent = "keke";
