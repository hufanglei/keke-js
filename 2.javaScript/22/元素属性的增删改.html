元素属性的增删改

	标签名称 tagName localName
	元素中的属性:自定义属性和 原始属性
	自定义属性
	getAttribute(key) 获取属性值
	setAttribute(key,value) 修改属性值
	removeAttribute(key) 删除属性值 //不是删除原始属性
	hasAttribute(key) 判断属性是否存在

		//2：自定属性
		//①getAttribute(key) 获取属性值
		function getAttr(key){
			var dom = document.getElementById("box");
			return  dom.getAttribute(key);
		}
		//var username = boxDom.getAttribute("username");
		var username = getAttr("username");
		var age = getAttr("age");
		var price = getAttr("price");
		var title = getAttr("title");

		//alert(username+"=="+age+"=="+price+"=="+title+"=="+id);

		//②修改属性值 
		function changeAttr(){
			var boxDom = dom("box");
			var value = dom("change").value;
			//var username = getAttr("username");
			boxDom.setAttribute("username" , value);
		}

		//removeAttribute(key) 删除属性值
		function removeAttr(){
			var boxDom = dom("box");
			var value = dom("change").value;
			boxDom.removeAttribute(value);
		}
