1.对象:object
arr是一个容器，装任何的东西，不能够很好的去体现一个事物完整的特性和信息。获取不方便维护，修改很麻烦。
对象是由键值对组成的一种数据形态。
格式: 
		var obj = {
			key:value,
			key1:value1,
			key2:value
		};
a:对象结尾的键值不要加逗号的.因为在ie6 7 8它会把它当做一种错误会阻断代码的执行。
b:第一个反应就检查对象的最后一个键值对有没有加逗号.

2.定义对象两种方式：var obj = {};或者 var obj = new Object(); 
	var person = {
		"username":"keke",
		"password":"456465"
	};
	var person = {//推荐使用第二种(除非是对象属性，对象属性加引号)
		username:"keke",
		password:"545647654"
	};
	var person = {
		'username':"sdfsdfsd",
		'password':"23423423"
	};
   
	//带对象属性的对象声明
	var person = {

3.对象修改，删除，新增，查找，属性探测，获取，枚举（循环） 
//修改
person["username"] = "spring";
person.username = "JXY";
person["user.email"] = "56464654@qq.com";

//获取
alert(person.username+"===="+person["username"]);
alert(person["user.email"]);

//删除属性和删除对象
//person = null;//删除对象
//删除属性
//delete person.username;

//检查元素是否存在对象 in,在就返回true否则返回false
//in有个特征就是不管是对象自定义属性还是父类object比如:toString,constructor,valueOf中继承的属性都可以识别。
//hasOwnProperty:检查属性是否存在对象中，但是不能检查父类中的属性.
alert("username" in person);//true
alert("age" in person);//true
alert("email" in person);//true
alert("toString" in person);//true
alert("constructor" in person);//true
alert("valueOf" in person);//true
alert(person.hasOwnProperty("username"));//false
alert(person.hasOwnProperty("age"));//true
alert(person.hasOwnProperty("toString"));//false
