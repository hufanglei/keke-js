window.onload = function(){
	  var timeDom = document.getElementById("time");		
		setInterval(function(){
			var date =  new Date();
			var fullYear = date.getFullYear();
			var month =  date.getMonth() +1;
			var day= date.getDate();
			//时分秒
			var hour = date.getHours();
			var min = date.getMinutes();
			var sec = date.getSeconds();
			var time = date.getMilliseconds();
			//星期数
			var week = getWeek(date.getDay());//0是星期天
			var formate = fullYear+"年"+month+"月"+day+"日"+hour+"时"+min+"分"+sec+"秒"+time+"毫秒数，"+week;
			timeDom.innerHTML = formate;
		},500);
		function getWeek(week){
			switch(week){
				case 0:return "星期天";break;
				case 1:return "星期一";break;
				case 2:return "星期二";break;
				case 3:return "星期三";break;
				case 4:return "星期四";break;
				case 5:return "星期五";break;
				case 6:return "星期六";break;
				default:return "";break;
			}
		}
	}
