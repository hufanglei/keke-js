window.onload =  function(){
	//距离明年春节放假还有多少天。
	//当前的时间和下一年的时间
	var timeDom = document.getElementById("time");
	function start(dom,y,m,d){
		var beginTime = new Date();
		var endTime = new Date(y,m-1,d);
		//获得毫秒属于差(两个格林尼治时间相减)
		var time = endTime.getTime() - beginTime.getTime();
		//获得剩余秒
		var sec =  time/1000;
		//换算
		var dd = Math.floor(sec/60/60/24);//天数
		var hh = Math.floor(sec/60/60%24);//小时
		var mm =  Math.floor(sec/60%60);//分钟
		var ss = Math.floor(sec%60);//秒
		var html =   "距离【"+endTime.getFullYear()+"年】春节还有："+dd+"天"+hh+"小时"+mm+"分"+ss+"秒";
		dom.innerHTML = html;
	}
	
	setInterval(function(){
		start(timeDom,2016,2,8);
	},1000);
}
